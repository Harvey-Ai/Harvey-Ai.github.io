2009-04-02 | Algorithm

整个都用的STL,用了队列，然后用set维护里面的数值，便于删除。

当然也可以维护一个当前有序的队列，并将元素与下标关联，查找的时候直接二分就可以，但是删除要想想怎么做。

#include <iostream>
#include<list>
#include <set>
using namespace std;
int main()
{
    int a,n,ncase,result;
    scanf("%d",&ncase);
    while(ncase--)
    {
        result=1;
        set<int> t;
        list<int> s;
        scanf("%d",&n);
        while(n--)
        {
            scanf("%d",&a);
            if(t.find (a) != t.end ())
            {
                int temp=s.size ();
                if(temp > result)
                    result = temp;
                int ttemp;
                while((ttemp=s.front ()) != a)
                {
                    t.erase (ttemp);
                    s.pop_front ();
                }
                s.pop_front ();
                s.push_back (a);
            }
            else
            {
                s.push_back (a);
                t.insert (a);
            }
        }
        int temp=s.size ();
        if(temp > result)
            result = temp;
        printf("%d\n",result);
    }
    return 0;
}
