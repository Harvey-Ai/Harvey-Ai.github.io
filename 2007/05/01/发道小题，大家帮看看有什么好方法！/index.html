Give a positive integer N, you tell me how many positive integer A that satisfies A | N && A is not 1, that is to say N % A == 0 && A != 1.

InputSeveral test cases, Each has an integer N (1 < N < 20000); OutputFor each case output an integer which describle in the statement. Sample Input12Sample Output5也不知道有什么先进的方法，所以就遍历了，哈哈。

#include <iostream>
#include <math.h>
using namespace std;
int main()
{
    int n,k,cnt,max;
    while(cin>>n)
    {
        if(n==1)
        {
            cout<<0<<endl;
            continue;
        }
        if(n==2)
        {
            cout<<1<<endl;
            continue;
        }
        cnt=0;max=sqrt(n);
        if(max*max==n)
        {
            cnt++;
            max=max-1;
        }
        for(k=2;k<=max;k++)
        {
            if(n%k==0)
                cnt+=2;
        }
        cout<<cnt+1<<endl;
    }
    return 0;
}
