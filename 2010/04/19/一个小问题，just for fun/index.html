一个小问题，just for fun

2010-04-19 | Algorithm

题目：给一堆数，数目可能很大，这堆数中除有两个不是匹配的外，其他的都是两两相等，问找出这两个数？

输入形式如：1 2 3 5 6 6 5 3 1 7
输出：2 7

因为数目很大，决定了问题的复杂性。
要满足时间上的要求还算比较简单。但是要想满足空间上的复杂度还是相当困难的。

时间上是O(n)，空间上是O(1)的复杂度是可能的吗？答案是肯定的。

这种问题要求数学能力较强，ACM的赛场上基本不会出现（除非。。。。），所以just for fun~~

附
题目：http://acm.jlu.edu.cn/joj/showproblem.php?pid=2653

#include 

int main()
{
    int i, a, n, res;
    int res1[33], res2[33], bit[33];

    bit[0] = 1;
    for(i = 1;i < 32;i++)
        bit[i] = bit[i-1] << 1;
    while(scanf("%d", &n), n)
    {
        res = 0;
        memset(res1, 0, sizeof(res1));
        memset(res2, 0, sizeof(res2));

        n *= 2;
        while(n--)
        {
            scanf("%d", &a);
            res ^= a;
            for(i = 0;i < 32;i++)
            {
                if (a & bit[i])
                    res1[i] ^= a;
                else
                    res2[i] ^= a;
            }
        }
        for(i = 0;i < 32;i++)
            if (res & bit[i])
            {
                res1[i] > res2[i] ? printf("%d %d\n", res2[i], res1[i]) : printf("%d %d\n", res1[i], res2[i]);
                break;
            }
    }
    return 0;
}
